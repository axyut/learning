  local ref=$(git symbolic-ref --short HEAD 2> /dev/null)
  if [ -n "${ref}" ]; then
    if [ -n "$(git status --porcelain)" ]; then
      local gitstatuscolor='%F{red}'
      echo "${gitstatuscolor} ðŸ’€[${ref}]"
    else
      local gitstatuscolor='%F{green}'

    fi
  else
    echo ""
  fi
}
